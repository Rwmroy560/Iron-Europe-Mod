#Mali ask France for help
character_event = {
	id = iron_century.2
	desc = "EVTDESCiron_century.2"
	picture = GFX_roy_soldiers
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes


	option = { 
		name = "EVTAiron_century.2" 
	}

	option = { 
		name = "EVTBiron_century.2" 
		ai_chance = { factor = 100 }
		add_character_modifier = { modifier = france_help years = 5 }
		event_target:good_france = { 
			make_tributary = {
				who = ROOT
				tributary_type = puppet
				}
			}
		capital_scope = {
			holder_scope = {
				IF = {
					limit = {
						capital_scope = {
							is_occupied = no
						}
					}
					spawn_unit = {
						owner = THIS
						province = PREV
						leader = none
						troops = {
							heavy_infantry = { 2000 3700 }
							light_cavalry = { 750 1500 }
						}
						attrition = 1.0
						maintenance_multiplier = 0.1
						merge = yes
						disband_on_peace = yes
					}
				}
			}
		}
		clear_event_target = good_france
	}
}
#Join Ukraine in war against Russia in SMO

	picture = GFX_roy_nato
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes

	trigger = {
		is_tributary = { type = nato }
	}
	option = { 
		name = "EVTAiron_century.3" 
		ai_chance = { factor = 25 }
		ROOT = {
			prestige = 100
			piety = 150
			join_defender_wars = event_target:war_ukraine
			}
		clear_event_target = war_ukraine
	}
	option = { 
		name = "EVTBiron_century.3"
		ai_chance = { factor = 25 }
		add_character_modifier = { modifier = best_interests years = 3 }
		clear_event_target = war_ukraine 
	}
	option = { 
		name = "EVTCiron_century.3"
		ai_chance = { factor = 50 }
		custom_tooltip = { text = Ukrhelp } 
		transfer_scaled_wealth = {
			from = ROOT
			to = event_target:war_ukraine
			value = 0.5
			}
		event_target:war_ukraine = {
		capital_scope = {
			holder_scope = {
				IF = {
					limit = {
						capital_scope = {
							is_occupied = no
						}
					}
					spawn_unit = {
						owner = THIS
						province = PREV
						leader = none
						troops = {
							light_infantry = { 250 2000 }
							light_cavalry = { 250 500 }
							pikemen = { 300 500 }
							archers = { 150 500 }
						}
						attrition = 1.0
						maintenance_multiplier = 0.1
						merge = yes
						disband_on_peace = yes
						reinforces = yes
						reinforce_rate_multiplier = .25
					}
				}
			}
		}
	}
	clear_event_target = war_ukraine 
	}
}
